class Solution {
public:
	int maxArea(vector<int>& height) {
		if (height.size() < 2) return 0;
		int result = 0;
		for (int i = 0, k = height.size() - 1; i < k;){
			result = std::max(result, std::min(height[i], height[k]) * (k - i));
			if (height[i] < height[k]) i++;
			else k--;
		}
		return result;
	}
};
