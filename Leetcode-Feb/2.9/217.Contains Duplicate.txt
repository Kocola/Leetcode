class Solution {
public:
	bool containsDuplicate(vector<int>& nums) {
		if (nums.size() < 1) return false;

		std::unordered_map<int, int> hashTable;
		for (int i = 0; i < nums.size(); ++i){
			if (hashTable.find(nums[i]) != hashTable.end()){
				return true;
			} else{
				hashTable[nums[i]] = 1;
			}
		}

		return false;
	}
};
