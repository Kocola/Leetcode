class Solution {
public:
	int searchInsert(vector<int>& nums, int target) {
		return lower_bound(nums, target);
	}
private:
	int lower_bound(const std::vector<int>& nums, int target){
		int lower = 0, upper = nums.size();
		while (lower < upper){
			int mid = lower + ((upper - lower) >> 1);
			if (nums[mid] < target){
				lower = mid + 1;
			} else{
				upper = mid;
			}
		}
		return lower;
	}
};
