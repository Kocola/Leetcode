class Solution {
public:
	int minMoves2(vector<int>& nums) {
		if (nums.size() < 2) return 0;
		std::sort(nums.begin(), nums.end());
		int result = 0, median = nums[(nums.size() - 1) / 2];
		for (int i = 0; i < nums.size(); ++i){
			result += std::abs(nums[i] - median);
		}
		return result;
	}
};
