class Solution {
public:
	string countAndSay(int n) {
		if (n < 3) return std::string(n, '1');
		std::string str = countAndSay(n - 1);
		int count = 1;
		std::string result;
		for (int i = 0; i < str.size() - 1; ++i){
			if (str[i] == str[i + 1]){
				count++;
			} else{
				result += std::string(1, count + '0') + str[i];
				count = 1;
			}
		}
		result += std::string(1, count + '0') + str.back();
		return result;
	}
};
