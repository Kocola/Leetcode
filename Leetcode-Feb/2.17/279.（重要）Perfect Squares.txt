class Solution {
	/* dp[i] = min(dp[i-j*j]), 1<= j*j <= i */
public:
	int numSquares(int n) {
		if (n <= 0) return 0;
		std::vector<int> dp(n + 1, 0);
		for (int i = 1; i <= n; ++i){
			int min = std::numeric_limits<int>::max();
			for (int k = 1; k * k <= i; ++k){
				min = std::min(min, dp[i - k * k] + 1);
			}
			dp[i] = min;
		}
		return dp[n];
	}
};
