class NumArray {
public:
	NumArray(vector<int> nums) {
		if (nums.empty()) return;
		sums.resize(nums.size() + 1, 0);
		for (int i = 0; i < nums.size(); ++i){
			sums[i + 1] = sums[i] + nums[i];
		}
	}

	int sumRange(int i, int j) {
		if (i > j) return 0;
		return sums[j + 1] - sums[i];
	}
private:
	std::vector<int> sums;
};
