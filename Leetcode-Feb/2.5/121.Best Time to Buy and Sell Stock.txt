class Solution {
public:
	int maxProfit(vector<int>& prices) {
		if (prices.size() == 0) return 0;
		int cur_min = prices[0];
		int result = 0;
		for (int i = 1; i < prices.size(); ++i){
			result = std::max(result, prices[i] - cur_min);
			cur_min = std::min(cur_min, prices[i]);
		}
		return result;
	}
};
